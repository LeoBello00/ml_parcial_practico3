import pandas as pd
from scipy.stats import chi2_contingency

col2drop = set()

df_fraud_str = df_fraud.fillna('NaN')

for i, col1 in enumerate(object_columns):
    for col2 in object_columns[i+1:]:
        if col2 not in col2drop:
            observed = pd.crosstab(df_fraud_str[col1], df_fraud_str[col2])
            # print(col1, col2)
            try:
                _, p, _, _ = chi2_contingency(observed)
            except ValueError:
                continue
            if p < 0.05:
                col2drop.add(col2)
                print(f"{col2} to drop with p={round(p, 2)}")
